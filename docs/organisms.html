YaroModal yaro-modal

Diálogo modal con overlay. Se renderiza con @if (open) — cuando open es false no existe en el DOM. Cierra con clic en el backdrop, botón ✕ y tecla Escape.

Confirmar eliminación
¿Estás segura de que deseas eliminar este producto? Esta acción no se puede deshacer.
size: sm closable: true closeOnBackdrop: true ESC: true 
// En el template
<yaro-modal
  [open]="showModal"
  title="Confirmar eliminación"
  size="sm"
  (closed)="showModal = false"
>
  <p>¿Estás segura? Esta acción no se puede deshacer.</p>

  <ng-container modal-actions>
    <yaro-button variant="secondary" (click)="showModal = false">Cancelar</yaro-button>
    <yaro-button variant="primary"   (click)="confirmar()">Eliminar</yaro-button>
  </ng-container>
</yaro-modal>

// En el componente
export class MiComponent {
  showModal = false;

  confirmar() {
    // lógica de confirmación
    this.showModal = false;
  }
}

// Modal de pantalla completa
<yaro-modal [open]="open" title="Editor" size="full" (closed)="open=false">
  <app-editor />
</yaro-modal>

YaroDataTable yaro-data-table

Tabla de datos con columnas configurables, ordenamiento client-side, paginación y estado de carga con skeleton. Soporta columnas de tipo texto, número, badge y delta. Las filas pueden ser clickeables.

Con paginación (pageSize=4)

// Definición de columnas
import { TableColumn } from '@yaro/ui';

columns: TableColumn[] = [
  { key: 'nombre',    header: 'Producto',  type: 'text'  },
  { key: 'stock',     header: 'Stock',     type: 'text', sortable: true, align: 'right' },
  { key: 'variacion', header: 'Variación', type: 'delta' },
  {
    key: 'estado', header: 'Estado', type: 'badge',
    intentFn: row => row['stock'] > 30 ? 'green' : row['stock'] > 10 ? 'amber' : 'red'
  },
];

page     = 1;
pageSize = 10;

// En el template
<yaro-data-table
  [columns]="columns"
  [rows]="productos"
  [loading]="cargando"
  [pageSize]="pageSize"
  [page]="page"
  [rowClickable]="true"
  (rowClick)="abrirDetalle($event)"
  (pageChange)="page = $event"
/>

// Sin paginación (por defecto) — muestra todas las filas
<yaro-data-table
  [columns]="columns"
  [rows]="productos"
/>

import { SidenavSection } from '@yaro/ui';

sections: SidenavSection[] = [
  {
    label: 'Principal',
    items: [
      { id: 'dashboard',  label: 'Dashboard',  icon: '🏠' },
      { id: 'ventas',     label: 'Ventas',     icon: '💰', badge: '12' },
      { id: 'inventario', label: 'Inventario', icon: '📦' },
    ],
  },
];

collapsed   = false;
mobileNavOpen = false;

// En el template — un solo bloque, funciona en desktop Y móvil
<yaro-topbar
  title="Mi App"
  (menuToggle)="mobileNavOpen = true"
/>

<yaro-sidenav
  [sections]="sections"
  [collapsed]="collapsed"
  [mobileOpen]="mobileNavOpen"
  (itemSelect)="navigate($event)"
  (toggleCollapse)="collapsed = !collapsed"
  (mobileClose)="mobileNavOpen = false"
/>

// El componente hace todo lo demás automáticamente:
// — Desktop: sidebar fija con collapse opcional
// — Móvil: drawer deslizante con overlay oscuro
//   (el overlay cierra el drawer al tocarlo)

YaroTopbar yaro-topbar

Barra superior sticky (z-index: 200). En móvil muestra automáticamente un botón hamburguesa (☰) que emite menuToggle — el desarrollador lo conecta al mobileOpen del sidenav. El slot topbar-center se oculta en pantallas pequeñas.

mobileNavOpen = false;

<yaro-topbar
  title="Inventario"
  subtitle="Bodega principal"
  (menuToggle)="mobileNavOpen = true"
>
  <ng-container topbar-actions>
    <yaro-connection-badge status="online" />
    <yaro-button variant="primary" size="sm">+ Nuevo</yaro-button>
  </ng-container>
</yaro-topbar>

// topbar-center se oculta automáticamente en móvil
<yaro-topbar title="Dashboard" (menuToggle)="mobileNavOpen = true">
  <ng-container topbar-center>
    <app-search-bar /> // visible solo en ≥768px
  </ng-container>
</yaro-topbar>

YaroBienestarPicker yaro-bienestar-picker

Selector de estado de bienestar del colaborador. 5 opciones predefinidas con emoji, color y label. Compatible con formularios vía ControlValueAccessor. Las opciones son completamente personalizables.

◈
ControlValueAccessor — usa [(ngModel)] o [formControl]. El valor emitido es el id de la opción seleccionada ('excelente', 'bien', etc.).
Opciones predefinidas (BIENESTAR_OPTIONS)

🙂Bien

😤Mal
// Con ngModel
<yaro-bienestar-picker [(ngModel)]="estadoBienestar" />

// Con Output manual
<yaro-bienestar-picker
  (selectedChange)="onBienestar($event)"
/>

// Opciones personalizadas
import { BienestarOption } from '@yaro/ui';

misopciones: BienestarOption[] = [
  { id: 'productivo', label: 'Productivo', icon: '🚀', color: 'green'  },

<yaro-bienestar-picker
  [options]="misopciones"

YaroConnectionBadge yaro-connection-badge

Indicador de estado de conexión con animación. Online pulsa suavemente, connecting y error parpadean. El punto cambia de color y animación automáticamente según el status. Tiene role="status" para accesibilidad.

// Estados básicos
<yaro-connection-badge status="online"     />
<yaro-connection-badge status="connecting" />
<yaro-connection-badge status="error"      />
<yaro-connection-badge status="offline"    />

// Con nombre de servicio
<yaro-connection-badge status="online"  service="DIAN" />
<yaro-connection-badge status="error"   service="DIAN" />

// Solo punto, sin texto
<yaro-connection-badge status="online" [showLabel]="false" />

// Dinámico — estado desde signal
<yaro-connection-badge
  [status]="conexionDian()"
  service="DIAN"
/>
